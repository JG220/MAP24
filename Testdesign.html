pip install pillow from PIL import Image, ImageDraw def create_gradient_background(width, height, color1, color2, output_path): image = Image.new('RGB', (width, height)) draw = ImageDraw.Draw(image) for y in range(height): r = int(color1[0] + (color2[0] - color1[0]) * (y / height)) g = int(color1[1] + (color2[1] - color1[1]) * (y / height)) b = int(color1[2] + (color2[2] - color1[2]) * (y / height)) draw.line([(0, y), (width, y)], fill=(r, g, b)) image.save(output_path) # Example usage width, height = 1920, 1080 color1 = (255, 0, 0) # Red color2 = (0, 0, 255) # Blue output_path = 'gradient_background.png' create_gradient_background(width, height, color1, color2, output_path)
